using namespace std;

// 1. bool을 반환하는 함수 정의
// 짝수 묻는 방법
bool isEven(int number) {
    if (number % 2 == 0) {
        return true;   // 짝수면 true(참)를 돌려줌
    } else {
        return false;  // 아니면 false(거짓)를 돌려줌
    }
}

int main() {
    int my_num = 10;

    // 2. 함수 호출 및 리턴값 사용
    if (isEven(my_num)) { 
        // isEven이 return true를 했기 때문에 이 문장이 실행됨
        cout << my_num << "은(는) 짝수입니다!" << endl;
    } else {
        cout << my_num << "은(는) 홀수입니다!" << endl;
    }

    return 0; // 메인 함수도 운영체제에게 "프로그램 성공!"이라는 0을 반환함
}
